```rust
    async fn fetch_minute_data(
        &self,
        code: &str,
        market: &str,
    ) -> Result<Vec<MinuteData>, String> {
        let secid = Self::code_to_secid(code, market);

        #[derive(Deserialize)]
        struct RawResponse {
            data: Option<RawTrend>,
        }
        #[derive(Deserialize)]
        struct RawTrend {
            trends: Option<Vec<String>>,
        }

        let params = [
            ("secid", secid.as_str()),
            ("fields1", "f1,f2,f3,f4,f5,f6,f7,f8,f9,f10,f11,f12,f13"),
            ("fields2", "f51,f52,f53,f54,f55,f56,f57,f58,f59,f60,f61"),
            ("ndays", "1"),
        ];

        let resp = self
            .client
            .get(EASTMONEY_TREND_URL)
            .query(&params)
            .send()
            .await
            .map_err(|e| format!("Trend request failed: {:#}", e))?;

        let body: RawResponse = resp
            .json()
            .await
            .map_err(|e| format!("Trend parse failed: {}", e))?;

        let trends = body
            .data
            .and_then(|d| d.trends)
            .unwrap_or_default();

        // Format: "time,price,avg_price,volume,amount,turnover"
        let data: Vec<MinuteData> = trends
            .iter()
            .filter_map(|line| {
                let parts: Vec<&str> = line.split(',').collect();
                if parts.len() >= 3 {
                    Some(MinuteData {
                        time: parts[0].to_string(),
                        price: parts[1].parse::<f64>().unwrap_or(0.0),
                        avg_price: parts[2].parse::<f64>().unwrap_or(0.0),
                        volume: parts.get(3).and_then(|v| v.parse().ok()).unwrap_or(0),
                    })
                } else {
                    None
                }
            })
            .collect();

        Ok(data)
    }
```

```vue
<script setup lang="ts">
import { ref, watch, onMounted, onUnmounted, nextTick } from 'vue';
import { init, dispose } from 'klinecharts';
import type { Chart, Period } from 'klinecharts';
import type { MinuteData } from '@/types';

const props = defineProps<{
  data: MinuteData[];
  name: string;
}>();

const chartContainer = ref<HTMLDivElement>();
let chart: Chart | null = null;

function buildChart() {
  if (!chartContainer.value || !props.data.length) return;

  if (chart) {
    dispose(chartContainer.value);
  }

  chart = init(chartContainer.value, {
    styles: {
      grid: { horizontal: { show: false }, vertical: { show: false } },
      candle: { bar: { upColor: '#f85149', downColor: '#3fb950' } },
      xAxis: { show: false },
      yAxis: { show: true, position: 'left' },
      indicator: { show: false },
      separator: { show: false }
    },
  });

  // Convert MinuteData to klinecharts format
  const klineData = props.data.map((d, i) => ({
    timestamp: i, // sequential index since we only have time strings
    open: d.price,
    close: d.price,
    high: d.price,
    low: d.price,
    volume: d.volume,
  }));

  chart.applyNewData(klineData as any);
}

onMounted(() => {
  nextTick(buildChart);
});

watch(() => props.data, () => {
  nextTick(buildChart);
}, { deep: true });

onUnmounted(() => {
  if (chartContainer.value && chart) {
    dispose(chartContainer.value);

});
```

```vue
<script setup lang="ts">
import { ref, watch } from 'vue';
import { invoke } from '@tauri-apps/api/core';
import { useQuoteStore } from '@/stores/quote';
import type { WatchItem, Depth, MinuteData } from '@/types';
import MinuteChart from './MinuteChart.vue';
import DepthPanel from './DepthPanel.vue';
import StockSummary from './StockSummary.vue';

const props = defineProps<{
  stock: WatchItem | null;
}>();

defineEmits<{
  close: [];
}>();

const quoteStore = useQuoteStore();

const depth = ref<Depth | null>(null);
const minuteData = ref<MinuteData[]>([]);
const loading = ref(false);

async function loadDetail() {
  if (!props.stock) return;
  loading.value = true;
  const { code, market } = props.stock;

  const [d, m] = await Promise.allSettled([
    invoke<Depth>('get_depth', { code, market }),
    invoke<MinuteData[]>('get_intraday', { code, market }),
  ]);

  if (d.status === 'fulfilled') depth.value = d.value;
  else depth.value = null;

  if (m.status === 'fulfilled') minuteData.value = m.value;
  else minuteData.value = [];

  loading.value = false;
}

watch(() => props.stock, (s) => {
  if (s) loadDetail();
}, { immediate: true });
</script>

        >
          {{ (quoteStore.getQuote(stock.code, stock.market)?.change_pct ?? 0) >= 0 ? '+' : '' }}{{ quoteStore.getQuote(stock.code, stock.market)?.change_pct?.toFixed(2) ?? '--' }}%
```

```rust
    /// Convert code to Tencent format: sh600519, sz000001
    fn code_to_tencent(code: &str, market: &str) -> String {
        if market == "CN" {
            if code.starts_with("6") || code.starts_with("5") || code.starts_with("9") {
                format!("sh{}", code)
            } else {
                format!("sz{}", code)
            }
        } else {
            code.to_string()
        }

    /// Parse Tencent quote line: v_sh600519="1~平安银行~000001~12.50~..."
    fn parse_tencent_line(line: &str) -> Option<Quote> {
        let eq_pos = line.find('=')?;
        let quote_start = line[eq_pos + 1..].find('"')? + eq_pos + 2;
        let quote_end = line[quote_start..].find('"')?;
        let data = &line[quote_start..quote_start + quote_end];
        let fields: Vec<&str> = data.split('~').collect();

        if fields.len() < 40 {
            return None;
        }

        // Fields reference (Tencent API):
        // 0: unknown, 1: name, 2: code, 3: price, 4: prev_close, 5: open,
        // 6: volume (lots), 7: bid_volume, 8: ask_volume, 9: bid1_price,
        // ... 31: change, 32: change_pct, 33: high, 34: low,
        // 35: price, 36: volume, 37: turnover (wan), 38: turnover_rate

        let code = fields.get(2).map(|s| s.to_string()).unwrap_or_default();
        let name = fields.get(1).map(|s| s.to_string()).unwrap_or_default();
        let price = fields.get(3).and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
        let prev_close = fields.get(4).and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
        let open = fields.get(5).and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
        let high = fields.get(33).and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
        let low = fields.get(34).and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
        let volume = fields.get(6).and_then(|s| s.parse::<u64>().ok()).unwrap_or(0);
        let turnover = fields.get(37).and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
        let turnover_rate = fields.get(38).and_then(|s| s.parse::<f64>().ok());
        let change = price - prev_close;
        let change_pct = if prev_close != 0.0 {
            (change / prev_close) * 100.0
        } else {
            0.0
        };

        Some(Quote {
            code,
            market: "CN".to_string(),
            name,
            price,
            change: (change * 100.0).round() / 100.0,
            change_pct: (change_pct * 100.0).round() / 100.0,
            open,
            high,
            low,
            volume,
            turnover,
            turnover_rate,
            timestamp: chrono::Utc::now().timestamp(),
        })
    }
```

```rust
use chrono::{Local, Datelike, NaiveTime, Weekday};

/// Market trading session
#[derive(Debug, PartialEq)]
pub enum MarketSession {
    PreOpen,      // Before 9:30
    MorningTrade, // 9:30 - 11:30
    LunchBreak,   // 11:30 - 13:00
    AfternoonTrade, // 13:00 - 15:00
    AfterClose,   // After 15:00 (weekday)
    Weekend,      // Saturday/Sunday
}

/// Determine the current A-share market session
pub fn current_session() -> MarketSession {
    let now = Local::now();
    let weekday = now.weekday();

    if weekday == Weekday::Sat || weekday == Weekday::Sun {
        return MarketSession::Weekend;
    }

    let time = now.time();

    let morning_open = NaiveTime::from_hms_opt(9, 30, 0).unwrap();
    let morning_close = NaiveTime::from_hms_opt(11, 30, 0).unwrap();
    let afternoon_open = NaiveTime::from_hms_opt(13, 0, 0).unwrap();
    let afternoon_close = NaiveTime::from_hms_opt(15, 0, 0).unwrap();

    if time < morning_open {
        MarketSession::PreOpen
    } else if time < morning_close {
        MarketSession::MorningTrade
    } else if time < afternoon_open {
        MarketSession::LunchBreak
    } else if time < afternoon_close {
        MarketSession::AfternoonTrade
    } else {
        MarketSession::AfterClose
    }
}

/// Get the recommended polling interval (in seconds) for the current session
pub fn recommended_interval(session: &MarketSession) -> u64 {
    match session {
        MarketSession::MorningTrade | MarketSession::AfternoonTrade => 2,
        MarketSession::LunchBreak => 10,
        MarketSession::PreOpen | MarketSession::AfterClose => 30,
        MarketSession::Weekend => 60,
    }
}
```
